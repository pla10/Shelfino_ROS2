166 protected:
167  static
168  void ros1_callback(
169  const ros::MessageEvent<ROS1_T const> & ros1_msg_event,
170  rclcpp::PublisherBase::SharedPtr ros2_pub,
171  const std::string & ros1_type_name,
172  const std::string & ros2_type_name,
173  rclcpp::Logger logger)
174  {
175  typename rclcpp::Publisher<ROS2_T>::SharedPtr typed_ros2_pub;
176  typed_ros2_pub =
177  std::dynamic_pointer_cast<typename rclcpp::Publisher<ROS2_T>>(ros2_pub);
178 
179  if (!typed_ros2_pub) {
180  throw std::runtime_error(
181  "Invalid type " + ros2_type_name + " for ROS 2 publisher " +
182  ros2_pub->get_topic_name());
183  }
184 
185  const boost::shared_ptr<ros::M_string> & connection_header =
186  ros1_msg_event.getConnectionHeaderPtr();
187  if (!connection_header) {
188  RCLCPP_WARN(
189  logger, "Dropping ROS 1 message %s without connection header", ros1_type_name.c_str());
190  return;
191  }
192 
193  std::string key = "callerid";
194  if (connection_header->find(key) != connection_header->end()) {
195  if (connection_header->at(key) == "/ros_bridge") {
196  return;
197  }
198  }
199 
200  const boost::shared_ptr<ROS1_T const> & ros1_msg = ros1_msg_event.getConstMessage();
201 
202  auto ros2_msg = std::make_unique<ROS2_T>();
203  convert_1_to_2(*ros1_msg, *ros2_msg);
204  RCLCPP_INFO_ONCE(
205  logger, "Passing message from ROS 1 %s to ROS 2 %s (showing msg only once per type)",
206  ros1_type_name.c_str(), ros2_type_name.c_str());
207  typed_ros2_pub->publish(std::move(ros2_msg));
208  }
209 
210  static
211  void ros2_callback(
212  typename ROS2_T::SharedPtr ros2_msg,
213  const rclcpp::MessageInfo & msg_info,
214  ros::Publisher ros1_pub,
215  const std::string & ros1_type_name,
216  const std::string & ros2_type_name,
217  rclcpp::Logger logger,
218  rclcpp::PublisherBase::SharedPtr ros2_pub = nullptr)
219  {
220  if (ros2_pub) {
221  bool result = false;
222  auto ret = rmw_compare_gids_equal(
223  &msg_info.get_rmw_message_info().publisher_gid,
224  &ros2_pub->get_gid(),
225  &result);
226  if (ret == RMW_RET_OK) {
227  if (result) { // message GID equals to bridge's ROS2 publisher GID
228  return; // do not publish messages from bridge itself
229  }
230  } else {
231  auto msg = std::string("Failed to compare gids: ") + rmw_get_error_string().str;
232  rmw_reset_error();
233  throw std::runtime_error(msg);
234  }
235  }
236 
237  void * ptr = ros1_pub;
238  if (ptr == 0) {
239  RCLCPP_WARN_ONCE(
240  logger,
241  "Message from ROS 2 %s failed to be passed to ROS 1 %s because the "
242  "ROS 1 publisher is invalid (showing msg only once per type)",
243  ros2_type_name.c_str(), ros1_type_name.c_str());
244  return;
245  }
246 
247  ROS1_T ros1_msg;
248  convert_2_to_1(*ros2_msg, ros1_msg);
249  RCLCPP_INFO_ONCE(
250  logger, "Passing message from ROS 2 %s to ROS 1 %s (showing msg only once per type)",
251  ros2_type_name.c_str(), ros1_type_name.c_str());
252  ros1_pub.publish(ros1_msg);
253  }

273 template<class ROS1_T, class ROS2_T>
274 class ServiceFactory : public ServiceFactoryInterface
275 {
276 public:
277  using ROS1Request = typename ROS1_T::Request;
278  using ROS2Request = typename ROS2_T::Request;
279  using ROS1Response = typename ROS1_T::Response;
280  using ROS2Response = typename ROS2_T::Response;
281 
282  void forward_2_to_1(
283  ros::ServiceClient client, rclcpp::Logger logger, const std::shared_ptr<rmw_request_id_t>,
284  const std::shared_ptr<ROS2Request> request, std::shared_ptr<ROS2Response> response)
285  {
286  ROS1_T srv;
287  translate_2_to_1(*request, srv.request);
288  if (client.call(srv)) {
289  translate_1_to_2(srv.response, *response);
290  } else {
291  throw std::runtime_error("Failed to get response from ROS 1 service " + client.getService());
292  }
293  }
294 
295  bool forward_1_to_2(
296  rclcpp::ClientBase::SharedPtr cli, rclcpp::Logger logger,
297  const ROS1Request & request1, ROS1Response & response1,
298  int service_execution_timeout)
299  {
300  auto client = std::dynamic_pointer_cast<rclcpp::Client<ROS2_T>>(cli);
301  if (!client) {
302  RCLCPP_ERROR(logger, "Failed to get ROS 2 client %s", cli->get_service_name());
303  return false;
304  }
305  auto request2 = std::make_shared<ROS2Request>();
306  translate_1_to_2(request1, *request2);
307  while (!client->wait_for_service(std::chrono::seconds(1))) {
308  if (!rclcpp::ok()) {
309  RCLCPP_ERROR(
310  logger, "Interrupted while waiting for ROS 2 service %s", cli->get_service_name());
311  return false;
312  }
313  RCLCPP_WARN(logger, "Waiting for ROS 2 service %s...", cli->get_service_name());
314  }
315  auto timeout = std::chrono::seconds(service_execution_timeout);
316  auto future = client->async_send_request(request2);
317  auto status = future.wait_for(timeout);
318  if (status == std::future_status::ready) {
319  auto response2 = future.get();
320  translate_2_to_1(*response2, response1);
321  } else {
322  RCLCPP_ERROR(
323  logger, "Failed to get response from ROS 2 service %s within %d seconds",
324  cli->get_service_name(), service_execution_timeout);
325  return false;
326  }
327  return true;
328  }

330  ServiceBridge1to2 service_bridge_1_to_2(
331  ros::NodeHandle & ros1_node, rclcpp::Node::SharedPtr ros2_node, const std::string & name,
332  int service_execution_timeout)
333  {
334  ServiceBridge1to2 bridge;
335  bridge.client = ros2_node->create_client<ROS2_T>(name);
336  auto m = &ServiceFactory<ROS1_T, ROS2_T>::forward_1_to_2;
337  auto f = std::bind(
338  m, this, bridge.client, ros2_node->get_logger(), std::placeholders::_1,
339  std::placeholders::_2, service_execution_timeout);
340  bridge.server = ros1_node.advertiseService<ROS1Request, ROS1Response>(name, f);
341  return bridge;
342  }

351  void(

353  const std::shared_ptr<ROS2Request>,
354  std::shared_ptr<ROS2Response>)> f;
355  f = std::bind(
356  m, this, bridge.client, ros2_node->get_logger(), std::placeholders::_1,
357  std::placeholders::_2, std::placeholders::_3);
358  bridge.server = ros2_node->create_service<ROS2_T>(name, f);
359  return bridge;
360  }
