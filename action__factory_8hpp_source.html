100  void goal_cb(ROS1GoalHandle gh1)
101  {
102  const std::string goal_id = gh1.getGoalID().id;
103 
104  // create a new handler for the goal
105  std::shared_ptr<GoalHandler> handler;
106  handler.reset(new GoalHandler(gh1, client_));
107  std::lock_guard<std::mutex> lock(mutex_);
108  goals_.insert(std::make_pair(goal_id, handler));
109 
110  RCLCPP_INFO(ros2_node_->get_logger(), "Sending goal");
111  std::thread(
112  [handler, goal_id, this]() mutable {
113  // execute the goal remotely
114  handler->handle();
115 
116  // clean-up
117  std::lock_guard<std::mutex> lock(mutex_);
118  goals_.erase(goal_id);

122 private:
123  class GoalHandler
124  {
125 public:
126  void cancel()
127  {
128  std::lock_guard<std::mutex> lock(mutex_);
129  canceled_ = true;
130  if (gh2_) { // cancel goal if possible
131  auto fut = client_->async_cancel_goal(gh2_);
132  }
133  }
134  void handle()
135  {
136  auto goal1 = gh1_.getGoal();
137  ROS2Goal goal2;
138  translate_goal_1_to_2(*gh1_.getGoal(), goal2);
139 
140  if (!client_->wait_for_action_server(std::chrono::seconds(1))) {
141  std::cout << "Action server not available after waiting" << std::endl;
142  gh1_.setRejected();
143  return;
144  }
145 
146  // goal_response_callback signature changed after foxy, this implementation
147  // works with both
148  std::shared_future<ROS2ClientGoalHandle> gh2_future;
149  // Changes as per Dashing
150  auto send_goal_ops = ROS2SendGoalOptions();
151  send_goal_ops.goal_response_callback =
152  [this, &gh2_future](std::shared_future<ROS2GoalHandle> gh2) mutable {
153  auto goal_handle = gh2_future.get();
154  if (!goal_handle) {
155  gh1_.setRejected(); // goal was not accepted by remote server
156  return;
157  }
158 
159  gh1_.setAccepted();
160 
161  {
162  std::lock_guard<std::mutex> lock(mutex_);
163  gh2_ = goal_handle;
164 
165  if (canceled_) { // cancel was called in between
166  auto fut = client_->async_cancel_goal(gh2_);
167  }
168  }
169  };
170 
171  send_goal_ops.feedback_callback = [this](ROS2GoalHandle, auto feedback2) mutable {
172  ROS1Feedback feedback1;
173  translate_feedback_2_to_1(feedback1, *feedback2);
174  gh1_.publishFeedback(feedback1);
175  };
176 
177  // send goal to ROS2 server, set-up feedback
178  gh2_future = client_->async_send_goal(goal2, send_goal_ops);
179 
180  auto future_result = client_->async_get_result(gh2_future.get());
181  auto res2 = future_result.get();
182 
183  ROS1Result res1;
184  translate_result_2_to_1(res1, *(res2.result));
185 
186  std::lock_guard<std::mutex> lock(mutex_);
187  if (res2.code == rclcpp_action::ResultCode::SUCCEEDED) {
188  gh1_.setSucceeded(res1);
189  } else if (res2.code == rclcpp_action::ResultCode::CANCELED) {
190  gh1_.setCanceled(res1);
191  } else {
192  gh1_.setAborted(res1);
193  }
194  }

197  : gh1_(gh1), gh2_(nullptr), client_(client), canceled_(false) {}

314 private:
315  class GoalHandler
316  {
317 public:
318  void cancel()
319  {
320  std::lock_guard<std::mutex> lock(mutex_);
321  canceled_ = true;
322  if (gh1_) {
323  // cancel goal if possible
324  gh1_->cancel();
325  }
326  }
327  void handle()
328  {
329  auto goal2 = gh2_->get_goal();
330  ROS1Goal goal1;
331  translate_goal_2_to_1(*goal2, goal1);
332 
333  if (gh2_->is_canceling()) {
334  auto result = std::make_shared<ROS2Result>();
335  gh2_->canceled(result);
336  return;
337  }
338 
339  std::condition_variable cond_result;
340  std::atomic<bool> result_ready(false);
341 
342  auto goal_handle = client_->sendGoal(
343  goal1,
344  [this, &result_ready,
345  &cond_result](ROS1ClientGoalHandle goal_handle) mutable // transition_cb
346  {
347  ROS_INFO("Goal [%s]", goal_handle.getCommState().toString().c_str());
348  if (goal_handle.getCommState() == actionlib::CommState::RECALLING) {
349  // cancelled before being processed
350  auto result2 = std::make_shared<ROS2Result>();
351  gh2_->canceled(result2);
352  return;
353  } else if (goal_handle.getCommState() == actionlib::CommState::ACTIVE) {
354  std::lock_guard<std::mutex> lock(mutex_);
355  gh1_ = std::make_shared<ROS1ClientGoalHandle>(goal_handle);
356  } else if (goal_handle.getCommState() == actionlib::CommState::DONE) {
357  auto result2 = std::make_shared<ROS2Result>();
358  auto result1 = goal_handle.getResult();
359  translate_result_1_to_2(*result2, *result1);
360  ROS_INFO("Goal [%s]", goal_handle.getTerminalState().toString().c_str());
361  if (goal_handle.getTerminalState() == actionlib::TerminalState::SUCCEEDED) {
362  gh2_->succeed(result2);
363  } else {
364  gh2_->abort(result2);
365  }
366  result_ready = true;
367  cond_result.notify_one();
368  return;
369  }
370  },
371  [this](ROS1ClientGoalHandle goal_handle, auto feedback1) mutable // feedback_cb
372  {
373  (void)goal_handle;
374  auto feedback2 = std::make_shared<ROS2Feedback>();
375  translate_feedback_1_to_2(*feedback2, *feedback1);
376  gh2_->publish_feedback(feedback2);
377  });
378 
379  std::mutex mutex_result;
380  std::unique_lock<std::mutex> lck(mutex_result);
381  cond_result.wait(lck, [&result_ready] {return result_ready.load();});
382  }
